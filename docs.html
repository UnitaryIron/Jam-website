Getting Started

Run Jam code directly in your browser:

Open Jam IDE

Variables

Declare variables with let:

let name = "Alice"
let age = 14
let is_coder = true

print name  # Output: Alice

Math Operations

add 5 and 3 into result
multiply 4 and 2 into product
square of 5  # Returns 25
sqrt of 49   # Returns 7

print result  # Output: 8

Lists

let fruits = ["apple", "banana", "cherry"]
choose from fruits into snack

print snack  # Output: "banana" (random)

Conditionals

let score = 85

if score > 90 {
    print "A"
} else if score > 80 {
    print "B"
} else {
    print "C"
}

Functions

function greet name {
    print "Hello, " + name
}

call greet "Jam"  # Output: Hello, Jam
